Sync to Google Sheets

Instantly synchronize your player stats bidirectionally. You can load statistics entered on your spreadsheet directly into this client dashboard, or export any manual changes back to Google Sheets instantly.

🔧 Developer tip: Pasting this URL writes it directly into your local database. To freeze it permanently for public viewers, set it inside the 'DEFAULT_SHEETS_API_URL' code variable!

How to connect:

  1. In your Google Sheet, open Extensions > Apps Script.
  2. Copy the script snippet shown on the right-hand panel.
  3. In Google Apps Script, delete default code, paste the script, and click Deploy > New Deployment.
  4. Select Web App, execute as Me, set access to Anyone.
  5. Deploy, copy the resulting Web App URL, and paste it in the configuration box above!

Apps Script Webhook Snippet

// Google Sheets Webhook Snippet - Version 5.1
function doGet() {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
    var lastRow = sheet.getLastRow();
    if (lastRow < 2) {
      return ContentService.createTextOutput(JSON.stringify([]))
        .setMimeType(ContentService.MimeType.JSON);
    }
    var data = sheet.getRange(2, 1, lastRow - 1, 11).getValues();
    var roster = data.map(function(row) {
      var timestampVal = new Date(0).toISOString();
      if (row[10]) {
        var parsedDate = new Date(row[10]);
        if (!isNaN(parsedDate.getTime())) timestampVal = parsedDate.toISOString();
      }
      return {
        name: row[0].toString(),
        id: row[1].toString(),
        power: Number(row[2]) || 0,
        merits: Number(row[3]) || 0,
        kills: Number(row[4]) || 0,
        donation: Number(row[5]) || 0,
        buildTime: Number(row[6]) || 0,
        destructionTime: Number(row[7]) || 0,
        help: Number(row[8]) || 0,
        assistance: Number(row[9]) || 0,
        timestamp: timestampVal
      };
    });
    return ContentService.createTextOutput(JSON.stringify(roster))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    var rawData = e.postData.contents;
    var parsedData = JSON.parse(rawData);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["Player Name", "Player ID", "Power", "Merits", "Kills", "Alliance Donation", "Build Time", "Destruction Time", "Alliance Help", "Resource Assistance", "Sync Timestamp"]);
    }
    if (sheet.getLastRow() > 1) {
      sheet.getRange(2, 1, sheet.getLastRow() - 1, 11).clearContent();
    }
    parsedData.forEach(function(member) {
      sheet.appendRow([
        member.name, member.id, Number(member.power), Number(member.merits), Number(member.kills), Number(member.donation) || 0, Number(member.buildTime) || 0, Number(member.destructionTime) || 0, Number(member.help) || 0, Number(member.assistance) || 0, member.timestamp || new Date().toISOString()
      ]);
    });
    return ContentService.createTextOutput(JSON.stringify({"status": "success", "rows": parsedData.length}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function onEdit(e) {
  try {
    var sheet = e.source.getActiveSheet();
    var range = e.range;
    if (range.getRow() > 1 && range.getColumn() >= 1 && range.getColumn() <= 10) {
      sheet.getRange(range.getRow(), 11).setValue(new Date().toISOString());
    }
  } catch(err) {}
}
